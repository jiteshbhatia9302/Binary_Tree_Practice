Create a binary tree of your choice containing 10 nodes using C++. Write functions to display the following traversals of the tree: Inorder traversal, Preorder traversal, Postorder traversal. Call each function one by one to display the tree traversals.
#include <iostream> using namespace std; class Node { public: int data; Node* left; Node* right; Node(int value) { data = value; left = NULL; right = NULL; } }; void inorder(Node* root) { if(root == NULL) return; inorder(root->left); cout << root->data << " "; inorder(root->right); } void preorder(Node* root) { if(root == NULL) return; cout << root->data << " "; preorder(root->left); preorder(root->right); } void postorder(Node* root) { if(root == NULL) return; postorder(root->left); postorder(root->right); cout << root->data << " "; } int main() { Node* root = new Node(10); root->left = new Node(20); root->right = new Node(30); root->left->left = new Node(44); root->left->right = new Node(25); root->right->left = new Node(16); root->right->right = new Node(72); root->left->left->left = new Node(85); root->left->left->right = new Node(19); root->right->right->left = new Node(60); cout << "Inorder Traversal: "; inorder(root); cout << endl; cout << "Preorder Traversal: "; preorder(root); cout << endl; cout << "Postorder Traversal: "; postorder(root); return 0; }
Output:
